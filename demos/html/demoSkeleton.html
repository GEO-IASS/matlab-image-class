Input image

% read input image
bin = Image.read('circles.png');

% display
show(bin);

Skeleton with 'thin'

% compute skeleton (same as "skeleton(bin, 'method', 'thin');")
skel = skeleton(bin);
figure;
show(skel);

% compute and display overlay
ovr = overlay(bin, skel);
figure;
show(ovr);

Skeleton with 'skel'

% compute skeleton
skel2 = skeleton(bin, 'method', 'skel');
figure;
show(skel2);

% compute and display overlay
ovr2 = overlay(bin, skel2);
figure;
show(ovr2);
